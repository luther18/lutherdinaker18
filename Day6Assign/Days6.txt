3. Write a program that takes a input and determines if the number is a prime.

echo  "Enter Numbers : "   #taking input from user
read n                    #reading num
for((i=2; i<=$n/2; i++))
do
  ans=$(( n%i ))
  if [ $ans -eq 0 ]
  then
    echo "$n is not a prime number."
    exit 0
  fi
done
echo "$n is a prime number."
