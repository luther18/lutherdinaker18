do
    first=$((${array[i]}))
    for ((j=$(($i+1));j<$length;j++))
    do
        second=$((${array[j]}))
        for ((k=$(($j+1));k<$length;k++))
        do
            third=$((${array[k]}))
            sum=$(($first+$second+$third))
            if [ $sum -eq 0 ]
            then
                echo "sum of " $first $second $third "is equal to Zero. "
            fi
        done
    done
done
